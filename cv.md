```JS
function scrollToElement(el) {
	// scroll to element css prop
	const section = document.querySelector(el);
	const sectionOffset = section.offsetTop;

	window.scrollTo({
		top: sectionOffset,
		left: 0,
		behavior: "smooth",
	});
}
```
